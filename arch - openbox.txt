## drivers and config programs

## openbox
yay -S openbox obconf obkey nitrogen picom tint2 rofi
? yay -S openbox-themes lxappearance

cp /etc/X11/xinit/xinitrc ~/.xinitrc
nano ~/.xinitrc {
    Delete last 5 lines (twm..)
    Add {
        picom &
        nitrogen --restore &
        tint2 &
        exec openbox
    }
}
